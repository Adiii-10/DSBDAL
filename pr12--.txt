import java.io.IOException;

import org.apache.hadoop.conf.Configuration;

import org.apache.hadoop.fs.Path;

import org.apache.hadoop.io.IntWritable;

import org.apache.hadoop.io.Text;

import org.apache.hadoop.mapreduce.Job;

import org.apache.hadoop.mapreduce.Mapper;

import org.apache.hadoop.mapreduce.Reducer;

import org.apache.hadoop.mapreduce.lib.input.FileInputFormat;

import org.apache.hadoop.mapreduce.lib.output.FileOutputFormat;

public class LogAnalysis {

    // --------------------------------------------------------
    // Mapper Class
    // --------------------------------------------------------

    public static class LogMapper
        extends Mapper<Object, Text,
                        Text, IntWritable> {

        private final static IntWritable one =
            new IntWritable(1);

        private Text logType = new Text();

        public void map(
            Object key,
            Text value,
            Context context
        ) throws IOException, InterruptedException {

            String line = value.toString();

            // Check log levels

            if (line.contains("ERROR")) {

                logType.set("ERROR");

                context.write(logType, one);
            }

            else if (line.contains("WARNING")) {

                logType.set("WARNING");

                context.write(logType, one);
            }

            else if (line.contains("INFO")) {

                logType.set("INFO");

                context.write(logType, one);
            }
        }
    }

    // --------------------------------------------------------
    // Reducer Class
    // --------------------------------------------------------

    public static class LogReducer
        extends Reducer<Text, IntWritable,
                         Text, IntWritable> {

        public void reduce(
            Text key,
            Iterable<IntWritable> values,
            Context context
        ) throws IOException, InterruptedException {

            int sum = 0;

            // Count occurrences

            for (IntWritable val : values) {

                sum += val.get();
            }

            context.write(
                key,
                new IntWritable(sum)
            );
        }
    }

    // --------------------------------------------------------
    // Main Method
    // --------------------------------------------------------

    public static void main(String[] args)
        throws Exception {

        Configuration conf = new Configuration();

        Job job = Job.getInstance(
            conf,
            "log analysis"
        );

        job.setJarByClass(LogAnalysis.class);

        // Set Mapper and Reducer

        job.setMapperClass(LogMapper.class);

        job.setReducerClass(LogReducer.class);

        // Output types

        job.setOutputKeyClass(Text.class);

        job.setOutputValueClass(
            IntWritable.class
        );

        // Set input and output paths

        FileInputFormat.addInputPath(
            job,
            new Path(args[0])
        );

        FileOutputFormat.setOutputPath(
            job,
            new Path(args[1])
        );

        // Exit status

        System.exit(
            job.waitForCompletion(true)
            ? 0 : 1
        );
    }
}

# 📘 **Distributed Log File Processing using Hadoop MapReduce (Detailed Theory)**

## **Introduction**

In large-scale systems, log files are generated continuously to record system events such as errors, warnings, and informational messages. These logs can become very large and difficult to analyze using traditional methods.

A distributed approach using Hadoop MapReduce allows efficient processing of such large log files by dividing the task across multiple nodes.

In this practical, a MapReduce application is designed to analyze system log files and count occurrences of different log levels such as **ERROR, WARNING, and INFO**.

---

## **Objective of the Application**

* Process large log files in a distributed environment
* Identify and count different log levels
* Generate summarized output for analysis

---

The system uses:

### 🔹 **HDFS (Hadoop Distributed File System)**

* Stores log files in distributed manner
* Ensures fault tolerance

### 🔹 **MapReduce**

* Processes data in parallel
* Divides work into smaller tasks
