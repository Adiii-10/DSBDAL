import pandas as pd


df = pd.read_csv('Iris Dataset Prac (1,3.2,6,10).csv')


stats = df.groupby('Species').agg(['mean', 'std', 'min', 'max'])


print("--- Mean, Std, Min, Max for each Species ---")
print(stats)


percentiles = df.groupby('Species').quantile([0.25, 0.5, 0.75])


print("\n--- Percentiles (25%, 50%, 75%) ---")
print(percentiles)
